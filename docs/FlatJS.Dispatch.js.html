  var api         = {},

  api.publish = function(topic, args) {
    if (cache[topic]) {
      for (var i = 0; i < cache[topic].length; i++) {
        if (cache[topic][i]) {
          cache[topic][i].fn.apply(this, args);
        }
      }
    }
  }

  api.subscribe = function(topic, callback) {
    if(!cache[topic]){
      cache[topic] = [];
